```ts
import path from "node:path";
import { access } from "node:fs/promises";
import { deletePaseoWorktreeChecked, listPaseoWorktrees } from "../../utils/worktree.js";
import { getDb } from "./db.js";

type ReconcileLogger = {
  child(bindings: Record<string, unknown>): ReconcileLogger;
  info(...args: any[]): void;
  warn(...args: any[]): void;
};

type ReconcileOptions = {
  paseoHome: string;
  logger?: ReconcileLogger;
};

export async function runStartupReconciliation(opts: ReconcileOptions): Promise<void> {
  const log = opts.logger?.child({ module: "startup-reconcile" });
  const db = getDb();

  // 1. Crash recovery: any 'provisioning' row = crash mid-create
  //    Clean up worktree if it exists, then delete the row.
  const provisioning = await db.all<{ project_id: string; thread_id: string; worktree_path: string }[]>(
    "SELECT project_id, thread_id, worktree_path FROM threads WHERE status = 'provisioning'"
  );

  for (const row of provisioning) {
    log?.info({ event: "startup_reconcile_crash_recovery", projectId: row.project_id, threadId: row.thread_id, worktreePath: row.worktree_path }, "Cleaning up crashed provisioning thread");
    // Try to delete worktree if it exists on disk
    try {
      await access(row.worktree_path);
      // Exists — get project repoRoot to delete properly
      const project = await db.get<{ repo_root: string }>(
        "SELECT repo_root FROM projects WHERE project_id = ?",
        row.project_id
      );
      if (project) {
        await deletePaseoWorktreeChecked({
          cwd: project.repo_root,
          worktreePath: row.worktree_path,
          paseoHome: opts.paseoHome,
          allowDirty: true,
        }).catch((err) => {
          log?.warn({ err, worktreePath: row.worktree_path }, "Failed to delete crashed provisioning worktree");
        });
      }
    } catch {
      // worktree doesn't exist on disk — that's fine, just delete the row
    }
    await db.run(
      "DELETE FROM threads WHERE project_id = ? AND thread_id = ?",
      row.project_id,
      row.thread_id
    );
  }

  // 2. Orphan worktree cleanup: worktrees on disk with no DB row
  const projects = await db.all<{ project_id: string; repo_root: string }[]>(
    "SELECT project_id, repo_root FROM projects"
  );

  for (const project of projects) {
    let worktrees: { path: string }[];
    try {
      worktrees = await listPaseoWorktrees({ cwd: project.repo_root, paseoHome: opts.paseoHome });
    } catch {
      continue;
    }

    for (const wt of worktrees) {
      const normalizedPath = path.resolve(wt.path);
      const row = await db.get(
        "SELECT thread_id FROM threads WHERE project_id = ? AND worktree_path = ?",
        project.project_id,
        normalizedPath
      );
      if (!row) {
        log?.info({ event: "startup_reconcile_orphan", worktreePath: normalizedPath }, "Deleting orphaned worktree");
        await deletePaseoWorktreeChecked({
          cwd: project.repo_root,
          worktreePath: normalizedPath,
          paseoHome: opts.paseoHome,
          allowDirty: true,
        }).catch((err) => {
          log?.warn({ err, worktreePath: normalizedPath }, "Failed to delete orphaned worktree");
        });
      }
    }
  }
}
```

Key decisions:
- `access()` to check if worktree exists before attempting deletion (avoids errors on already-missing paths)
- Errors on individual worktree deletions are warned, not thrown — reconciliation continues
- No tmux/agent cleanup: that was reaper scope, not needed here (agents are ephemeral, tmux sessions die with the process)
