    async getActiveThread(): Promise<ThreadRecord | null> {
      await this.load();
      const { projectId, threadId } = this.state.active;
      if (!projectId || !threadId) {
        return null;
      }
      return this.state.threads.find(
        (t) => t.projectId === projectId && t.threadId === threadId
      ) ?? null;
    }
